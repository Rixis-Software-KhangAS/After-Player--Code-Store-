precision highp float;

varying vec2 vTexCoord;
uniform sampler2D u_texture;

void main() {
    vec2 uv = vTexCoord;
    
    // Phản chiếu ngang
    uv.x = 1.0 - uv.x;
    
    vec4 color = texture2D(u_texture, uv);
    gl_FragColor = color;
}
